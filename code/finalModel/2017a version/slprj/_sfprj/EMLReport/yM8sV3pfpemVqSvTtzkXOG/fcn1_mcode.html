1function dutycycles = fcn(Tmx,Tmy,Tmz,B,m)
 2%B should be in body frame!!
 3
 4Tm = [Tmx;Tmy;Tmz];
 5BcrossMatrix=[0,-B(3),B(2);B(3),0,-B(1);-B(2),B(1),0];
 6dutycycles=pinv(BcrossMatrix)*(-Tm)/m;
 7%should be restricted to 0-1
 8dutycycles(dutycycles>1.0) = 1.0; %pwm duty cycle should be 100%;
 9dutycycles(dutycycles<0.0) = 0.0;
10end
